title: "Processes"

<Card title="Endpoint Description" icon="info" horizontal>
Retrieves all current processes from a `root` scope
</Card>

<ParamField
  query="pid"
  type="Number"
>
  A process ID to filter with
</ParamField>

<ParamField
  query="command"
  type="String"
>
  A `execCommand` to filter with
</ParamField>
