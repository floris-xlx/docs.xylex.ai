<Card title="Endpoint Description" icon="info" horizontal>
  Retrieves logs for a specified project. Requires a valid `log_file_path` in the request body.
</Card>

<BodyField
  name="log_file_path"
  type="String"
  required
  children="The file path of the log file to retrieve (e.g., `/home/floris-xlx/.pm2/logs/suitsbooks.nl-error.log`)."
>
  The file path of the log file to retrieve.
</BodyField>
