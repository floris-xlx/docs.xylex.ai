<Card title="Endpoint Description" icon="info" horizontal >Checks if an organization name is unique and adds it to the database. Returns the organization name and a boolean indicating if the creation was successful.</Card>





<ParamField query="organization_name" type="String" required  children="This parameter specifies the name of the organization that is to be created or checked for uniqueness in the database.">
  The name of the organization.
</ParamField>
<ParamField query="organization_id" type="String" required  children="This parameter provides the unique identifier for the organization, used in the creation or uniqueness check process.">
  The ID of the organization.
</ParamField>
