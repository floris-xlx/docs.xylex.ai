<Card title="Endpoint Description" icon="info" horizontal >Fetches weekly trades for a specified algorithm, verifying the user`s organization. Returns trade data or appropriate error messages based on user validation and data retrieval success.</Card>





<ParamField query="user_id" type="String" required  children="This is a unique identifier for the user requesting the trades.">
  The ID of the user.
</ParamField>
<ParamField query="organization" type="String" required  children="This represents the name of the organization that the user belongs to or is associated with.">
  The organization of the user.
</ParamField>
<ParamField query="week_nr" type="Option<i64>"  children="This optional parameter specifies the week number for which the trades are being requested. If not provided, the current week is used.">
  The week number.
</ParamField>
<ParamField query="algorithm_id" type="String" required  children="This is a unique identifier for the algorithm used to fetch trades. It is required in the context of fetching trades related to a specific algorithm.">
  The ID of the algorithm.
</ParamField>
