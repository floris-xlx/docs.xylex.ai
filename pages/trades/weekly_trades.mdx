<Card title="Endpoint Description" icon="info" horizontal >Fetches weekly trades for a specific algorithm in an organization. Requires user ID, organization, and algorithm ID; week number is optional. Validates user organization and filters trades by status.</Card>





<ParamField query="user_id" type="String" required  children="The unique identifier for the user.">
  User ID
</ParamField>
<ParamField query="organization" type="String" required  children="The organization to which the user belongs.">
  Organization
</ParamField>
<ParamField query="week_nr" type="Option<i64>"  children="The specific week number for which trades are requested. If not provided, defaults to the current week.">
  Week Number
</ParamField>
<ParamField query="algorithm_id" type="String" required  children="The unique identifier for the algorithm associated with the trades.">
  Algorithm ID
</ParamField>
