<Card title="Endpoint Description" icon="info" horizontal >Fetches weekly trades for a given organization and algorithm, validating user organization before querying the database. Accepts user ID, organization, algorithm ID, and optional week number as query parameters. Returns trades excluding unapproved and invalid statuses.</Card>





<ParamField query="user_id" type="String" required  children="A unique identifier for the user making the request, required for verification and data retrieval.">
  User ID
</ParamField>
<ParamField query="organization" type="String" required  children="The name of the organization the user is associated with, required for data filtering and verification.">
  Organization name
</ParamField>
<ParamField query="week_nr" type="Option<i64>"  children="An optional parameter representing the week number for which trades are requested. Defaults to the current week if not provided.">
  Week number
</ParamField>
<ParamField query="algorithm_id" type="String" required  children="A unique identifier for the algorithm for which trades are being requested, required for filtering the trades.">
  Algorithm ID
</ParamField>
