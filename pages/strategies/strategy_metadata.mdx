<Card title="Endpoint Description" icon="info" horizontal >Fetches metadata for a strategy based on its strategy ID, returning a JSON response with the strategy`s status and details.</Card>





<ParamField query="strategy_id" type="String" required  children="A string representing the unique identifier for a specific strategy. This is used to fetch metadata related to that strategy.">
  The unique identifier for the strategy.
</ParamField>
<ParamField query="organization" type="String" required  children="A string representing the name of the organization. This is used to fetch metadata related to strategies associated with the organization.">
  The name of the organization.
</ParamField>
