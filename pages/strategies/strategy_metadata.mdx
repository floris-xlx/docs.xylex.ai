<Card title="Endpoint Description" icon="info" horizontal >Fetches strategy metadata using an organization ID from the database, returns the data along with a status indicating if the data is available.</Card>





<ParamField query="strategy_id" type="String" required  children="This parameter is used to fetch metadata associated with a specific strategy by its unique identifier.">
  The unique identifier for a strategy.
</ParamField>
<ParamField query="organization" type="String" required  children="This parameter is used to fetch metadata associated with strategies belonging to a specific organization.">
  The name or identifier of the organization.
</ParamField>
