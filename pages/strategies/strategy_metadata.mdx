<Card title="Endpoint Description" icon="info" horizontal >Fetches strategy metadata for a given organization ID. Logs the organization and checks the database for related metadata. Returns a JSON response with the status and data.</Card>





<ParamField query="strategy_id" type="String" required  children="This parameter is used to fetch strategy metadata based on the specific strategy's unique identifier.">
  The unique identifier for a strategy.
</ParamField>
<ParamField query="organization" type="String" required  children="This parameter is used to fetch strategy metadata associated with the specified organization.">
  The name of the organization.
</ParamField>
