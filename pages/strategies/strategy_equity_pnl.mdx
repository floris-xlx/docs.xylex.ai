<Card title="Endpoint Description" icon="info" horizontal >Fetches the equity P&L data for a specified strategy ID and time period, returning JSON with strategy status and computed P&L data.</Card>





<ParamField query="strategy_id" type="String" required  children="A unique identifier for the strategy whose equity PnL data is to be fetched.">
  The ID of the strategy
</ParamField>
<ParamField query="data_length" type="Option<String>"  children="Optional parameter specifying the time span for the data to be fetched. Defaults to '12_months' if not provided.">
  The length of data to fetch
</ParamField>
